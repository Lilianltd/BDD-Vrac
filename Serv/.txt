function getClientNom($request_info)
{
	global $conn;
	$Prenom = $_GET['Prenom'];
	$query = "SELECT Nom FROM `Client` WHERE `Actif/Passif` = 'Actif' AND Prenom = '".$Prenom."'";
	$response = array();
	$result = mysqli_query($conn, $query);
	while($row = mysqli_fetch_row($result))
	{
		$response[] = $row[0];
	}
	header('Content-Type: application/json');
	return json_encode($response);
}
